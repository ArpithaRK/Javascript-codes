/*Program to find local minimum and local maximum values. Given an integer array. How many local extrema are there in array.
//local extrema: both local maximum values and minimum values.
//the array element either it should be greater than or less than neighbours(left and right side).
//input1 :[1,2,1,-1,1,0,1]
->explaination : 2 is greater than  left side 1 and right side 1 so count=1
->-1 is less than left side 1 and right side 1 so count =2
->1 is greater than left side -1 and right side 0 so count =3
->0 is less than left side 1 and right side 1 so count =4
->output: count : 4
Example 2: input = [3,2,4,1,3,0,5,6]
output= 5*/

function localExtrema(arr){
    let count =0;
    for(let i=1;i<arr.length-1;i++){
        if((arr[i]>arr[i-1] && arr[i]>arr[i+1]) || (arr[i]<arr[i-1] && arr[i]<arr[i+1]))
        count++;
    }
    console.log("count of local extremas:", count)
}
let arr1 = [3,2,4,1,3,0,5,6]
localExtrema(arr1)
